Code Snippet 
$(function () {

<!-- This is to handle each direction -->
  $(".p1").resizable({    
    handles: "n, e, w, s, se, sw, ne, nw"
  });

<!-- To have Draggable grid effect -->
  $(".p2").draggable({
    
    grid: [20, 20]
  });

<!-- Takes id at runtime -->
  $(".square").each(function () {
   $(this).resizable({
        handles: $(this).attr("id")
   });
  });
})
