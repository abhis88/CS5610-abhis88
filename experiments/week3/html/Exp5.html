Code Snippet 
$(function () {

<!-- Slider -->
$("#slider-1").slider();

<!-- Slider with initial value-->
$("#slider-2").slider({
    value: 60,
    animate: "slow",
    orientation: "horizontal"
});

<!-- Slider min and max value -->
$("#slider-3").slider({
    range: true,
    min: 0,
    max: 500,
    values: [35, 200],
    slide: function (event, ui) {
        $("#price").val("$" + ui.values[0] + " - $" + ui.values[1]);
    }
});

<!-- Slider with min and max value displayed -->
$("#slider-6").slider({
    range: true,
    min: 0,
    max: 500,
    values: [35, 200],
    start: function (event, ui) {
        $("#startvalue")
            .val("$" + ui.values[0] + " - $" + ui.values[1]);
    },
    stop: function (event, ui) {
        $("#stopvalue")
            .val("$" + ui.values[0] + " - $" + ui.values[1]);
    },
    change: function (event, ui) {
        $("#changevalue")
            .val("$" + ui.values[0] + " - $" + ui.values[1]);
    },
    slide: function (event, ui) {
        $("#slidevalue")
            .val("$" + ui.values[0] + " - $" + ui.values[1]);
    }
});
});
